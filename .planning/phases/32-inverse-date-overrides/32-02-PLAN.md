```typescript
async function handleConfirm() {
  startTransition(async () => {
    const result = await commitInverseOverrideAction({
      override_date,
      isFullDayBlock: mode === "block",
      unavailableWindows: mode === "unavailable" ? windows : [],
      affectedBookingIds: affected.map((b) => b.id),
      reason: noteValue || undefined,
    });
    if ("quotaError" in result && result.quotaError) {
      // Refresh-displayed quota state if the count drifted between preview and confirm
      setRemainingQuota(result.remaining);
      setQuotaError(true);
      return;
    }
    if (!result.ok) {
      toast.error(result.formError ?? "Failed to save override");
      return;
    }
    toast.success(
      result.cancelledCount > 0
        ? `Saved. Cancelled ${result.cancelledCount} booking${result.cancelledCount === 1 ? "" : "s"}.`
        : "Saved.",
    );
    if (result.emailFailures.length > 0) {
      toast.warning(`${result.emailFailures.length} email(s) failed to send. Check the bookings page.`);
    }
    closeModal();
    router.refresh();
  });
}
```

```typescript
async function handleSave() {
  startTransition(async () => {
    const previewResult = await previewAffectedBookingsAction({...});
    if (!previewResult.ok) { toast.error(previewResult.error); return; }
    if (previewResult.affected.length === 0) {
      // Fast path: no cancels needed
      await upsertDateOverrideAction({...});
      toast.success("Saved.");
      closeModal();
      router.refresh();
      return;

    setRemainingQuota(previewResult.remainingQuota);
    setQuotaError(previewResult.affected.length > previewResult.remainingQuota);
    setState("preview-ready");
  });
}
```
