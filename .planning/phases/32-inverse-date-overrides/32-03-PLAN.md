```typescript
import { TZDate } from "@date-fns/tz";
import type { SupabaseClient } from "@supabase/supabase-js";

export interface AffectedBooking {
  id: string;
  start_at: string;       // UTC ISO
  end_at: string;
  booker_name: string;
  event_type_name: string;
}

/**
 * Phase 32: Find confirmed bookings on `overrideDate` whose start_at falls inside
 * any of the proposed `unavailableWindows`. Filtering is done in JS using the
 * account timezone — same TZDate pattern as countBookingsOnLocalDate in lib/slots.ts.
 *
 * @param supabase - server-side Supabase client
 * @param accountId - account UUID
 * @param overrideDate - "YYYY-MM-DD" in account timezone
 * @param unavailableWindows - windows the owner proposes to block (start/end_minute from local midnight)
 * @param accountTimezone - IANA tz (e.g., "America/Chicago")
 * @returns confirmed bookings whose start_at is inside any unavailable window, sorted chronologically
 */
export async function getAffectedBookings(
  supabase: SupabaseClient,
  accountId: string,
  overrideDate: string,
  unavailableWindows: Array<{ start_minute: number; end_minute: number }>,
  accountTimezone: string,
): Promise<AffectedBooking[]> {
  if (unavailableWindows.length === 0) return [];

  // Widen to a UTC day window. The local-day boundary in `accountTimezone` may
  // span ~26 UTC hours due to DST edges; widen by one extra hour each side and
  // filter in JS by exact local-time match.
  const startUtc = `${overrideDate}T00:00:00.000Z`;
  const endUtc = `${overrideDate}T23:59:59.999Z`;

  const { data, error } = await supabase
    .from("bookings")
    .select("id, start_at, end_at, booker_name, event_types!inner(name)")
    .eq("account_id", accountId)
    .eq("status", "confirmed")
    .gte("start_at", startUtc)
    .lte("start_at", endUtc);

  if (error) throw error;
  if (!data) return [];

  return data
    .filter((b) => {
      const localStart = new TZDate(new Date(b.start_at), accountTimezone);
      // Confirm we're on the override date in local TZ (handles DST edges)
      const localDate = `${localStart.getFullYear()}-${String(localStart.getMonth() + 1).padStart(2, "0")}-${String(localStart.getDate()).padStart(2, "0")}`;
      if (localDate !== overrideDate) return false;
      const startMinute = localStart.getHours() * 60 + localStart.getMinutes();
      return unavailableWindows.some(
        (w) => startMinute >= w.start_minute && startMinute < w.end_minute,
      );
    })
    .map((b) => ({
      id: b.id,
      start_at: b.start_at,
      end_at: b.end_at,
      booker_name: b.booker_name,
      event_type_name:
        (Array.isArray(b.event_types) ? b.event_types[0] : b.event_types)?.name ?? "Unknown",
    }))
    .sort((a, b) => a.start_at.localeCompare(b.start_at));
}
```

Server action body:

```typescript
export async function commitInverseOverrideAction(
  rawInput: unknown,
): Promise<CommitOverrideResult> {
  const parsed = inputSchema.safeParse(rawInput);
  if (!parsed.success) {
    return { ok: false, formError: parsed.error.errors[0]?.message ?? "Invalid input" };
  }
  const input = parsed.data;

  // 1. Auth: get { accountId, accountTimezone } from the existing session helper.
  //    Match how actions.ts in the same directory does this.
  const { accountId, accountTimezone, supabase } = await getOwnerSessionContext();
  if (!accountId) return { ok: false, formError: "Not authenticated" };

  // 2. Quota pre-flight (HARD gate per EMAIL-23).
  //    With skipOwnerEmail=true (Task 1), each cancelBooking() sends 1 email (booker only).
  //    Therefore: needed = affectedBookingIds.length.
  const remaining = await getRemainingDailyQuota();
  const needed = input.affectedBookingIds.length;
  if (needed > remaining) {
    return { ok: false, quotaError: true, needed, remaining };
  }

  // 3. Write override rows (delete-all-for-date + insert new shape — matches the
  //    existing upsertDateOverrideAction pattern in actions.ts; if that pattern is
  //    extracted into a helper, reuse it).
  await supabase.from("date_overrides").delete()
    .eq("account_id", accountId)
    .eq("override_date", input.override_date);

  if (input.isFullDayBlock) {
    await supabase.from("date_overrides").insert({
      account_id: accountId,
      override_date: input.override_date,
      is_closed: true,
      start_minute: null,
      end_minute: null,
      note: input.reason ?? null,
    });
  } else {
    await supabase.from("date_overrides").insert(
      input.unavailableWindows.map((w) => ({
        account_id: accountId,
        override_date: input.override_date,
        is_closed: false,
        start_minute: w.start_minute,
        end_minute: w.end_minute,
        note: input.reason ?? null,
      })),
    );
  }

  // 4. RACE-SAFETY: re-query affected bookings AFTER the override rows are written.
  //    The slot engine now blocks new bookings on this date; any booking that
  //    snuck in between the UI preview and now is also covered.
  const finalAffected = input.isFullDayBlock
    ? await getAllConfirmedBookingsOnDate(supabase, accountId, input.override_date, accountTimezone)
    : await getAffectedBookings(supabase, accountId, input.override_date, input.unavailableWindows, accountTimezone);

  // Use the union of preview IDs and freshly-discovered IDs to ensure no booking is missed.
  const idsToCancel = Array.from(
    new Set([...input.affectedBookingIds, ...finalAffected.map((b) => b.id)]),
  );

  // 5. Batch cancel via existing cancelBooking() lifecycle.
  const appUrl = process.env.NEXT_PUBLIC_APP_URL ?? "http://localhost:3000";
  const settled = await Promise.allSettled(
    idsToCancel.map((id) =>
      cancelBooking({
        bookingId: id,
        actor: "owner",
        reason: input.reason ?? "Owner marked this time as unavailable",
        appUrl,
        ip: null,
        skipOwnerEmail: true, // batch-initiated by owner; suppress N duplicate owner emails
      }),
    ),
  );

  let cancelledCount = 0;
  const emailFailures: Array<{ bookingId: string; error: string }> = [];
  settled.forEach((r, idx) => {
    if (r.status === "fulfilled" && r.value.ok) {
      cancelledCount += 1;
      // Some implementations of cancelBooking return { ok, emailFailed } — propagate
      if ("emailFailed" in r.value && r.value.emailFailed) {
        emailFailures.push({ bookingId: idsToCancel[idx], error: "email send failed" });
      }
    } else if (r.status === "rejected") {
      emailFailures.push({ bookingId: idsToCancel[idx], error: String(r.reason) });
    }
  });

  // 6. Revalidate the availability page so the override list re-renders.
  revalidatePath("/app/availability");
  revalidatePath("/app/bookings");

  return { ok: true, cancelledCount, emailFailures };
}
```
