The new function returns:
```ts
export interface ResolvedChromeColors {
  /** Direct hex fill for page (SidebarInset) bg. Null = use CSS default (bg-background / gray-50). */
  pageColor: string | null;
  /** Direct hex fill for sidebar bg. Null = use CSS default (--sidebar token). */
  sidebarColor: string | null;
  /** Direct hex for --primary CSS variable override. Null = use shadcn default. */
  primaryColor: string | null;
  /** WCAG text color for sidebar surface. Null when sidebarColor is null. */
  sidebarTextColor: "#ffffff" | "#000000" | null;
  /** WCAG text color for primary-colored surfaces (button text, etc.). Null when primaryColor is null. */
  primaryTextColor: "#ffffff" | "#000000" | null;
}

export function resolveChromeColors(branding: Branding): ResolvedChromeColors {
  const pageColor = branding.backgroundColor ?? null;
  const sidebarColor = branding.sidebarColor ?? null;
  // brand_primary is always a string in Branding (falls back to DEFAULT_BRAND_PRIMARY),
  // but we want null when the owner has not set it so shell can skip the CSS var override.
  // Check: if primaryColor === DEFAULT_BRAND_PRIMARY AND the original DB value was null,
  // we should not override --primary (that would force navy even if owner hasn't set brand_primary).
  // Solution: expose the raw value. Since Branding.primaryColor always has a fallback,
  // we cannot distinguish "owner set navy" from "default navy".
  // For Phase 12.6, accept that primaryColor is always set (defaults to DEFAULT_BRAND_PRIMARY).
  // Shell sets --primary to primaryColor unconditionally; this is safe because the default
  // value (#0A2540) is the same as the shadcn-overridden default we already use elsewhere.
  // v1.2 improvement: store raw brand_primary nullable separately in Branding.
  const primaryColor = branding.primaryColor; // always string, never null
  const sidebarTextColor = sidebarColor ? pickTextColor(sidebarColor) : null;
  const primaryTextColor = pickTextColor(primaryColor);

  return {
    pageColor,
    sidebarColor,
    primaryColor,
    sidebarTextColor,
    primaryTextColor,
  };
}
```
