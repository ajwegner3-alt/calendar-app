**File 1: `app/(shell)/app/billing/_components/subscribe-tier-card.tsx`**

```typescript
"use client";

import { useState, useTransition } from "react";
import { useRouter } from "next/navigation";
import { Loader2, Check } from "lucide-react";
import { Card, CardHeader, CardTitle, CardDescription, CardContent, CardFooter, CardAction } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Button } from "@/components/ui/button";
import type { PriceTier, PriceInterval } from "@/lib/stripe/prices";

interface SubscribeTierCardProps {
  tier: PriceTier;                              // 'basic' or 'widget'
  interval: PriceInterval;                      // controlled by parent TierGrid
  monthlyLabel: string;                         // e.g. "$29/month"
  annualMonthlyEquivalentLabel: string;         // e.g. "$23/month"
  annualTotalLabel: string;                     // e.g. "$278.40/year"
  savingsPct: number;                           // e.g. 20
  featured?: boolean;                           // true for Widget — adds "Most popular" badge
}

const TIER_COPY: Record<PriceTier, { title: string; description: string }> = {
  basic:  { title: "Basic",  description: "Booker access without the embeddable widget." },
  widget: { title: "Widget", description: "Full booking widget + everything in Basic." },
};

export function SubscribeTierCard({
  tier, interval,
  monthlyLabel, annualMonthlyEquivalentLabel, annualTotalLabel, savingsPct,
  featured = false,
}: SubscribeTierCardProps) {
  const router = useRouter();
  const [isPending, startTransition] = useTransition();
  const [error, setError] = useState<string | null>(null);

  const handleSubscribe = () => {
    setError(null);
    startTransition(async () => {
      try {
        const res = await fetch("/api/stripe/checkout", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ tier, interval }),
        });
        if (!res.ok) {
          const body = await res.json().catch(() => ({})) as { error?: string };
          setError(body.error ?? "checkout_failed");
          return;
        }
        const { url } = await res.json() as { url: string };
        if (url) {
          window.location.assign(url); // Hosted Checkout — same-window navigation
        } else {
          setError("no_redirect_url");
        }
      } catch {
        setError("network_error");
      }
    });
  };

  const priceLabel = interval === "monthly" ? monthlyLabel : annualMonthlyEquivalentLabel;
  const secondaryLabel = interval === "annual" ? `${annualTotalLabel} (save ${savingsPct}%)` : null;
  const copy = TIER_COPY[tier];

  return (
    <Card className={featured ? "border-primary shadow-md" : undefined} data-tier={tier}>
      <CardHeader>
        <CardTitle>{copy.title}</CardTitle>
        {featured ? (
          <CardAction>
            <Badge variant="default">Most popular</Badge>
          </CardAction>
        ) : null}
        <CardDescription>{copy.description}</CardDescription>
      </CardHeader>
      <CardContent>
        <p className="text-3xl font-semibold">{priceLabel}</p>
        {secondaryLabel ? (
          <p className="text-sm text-muted-foreground mt-1">{secondaryLabel}</p>
        ) : null}
      </CardContent>
      <CardFooter className="flex-col items-stretch gap-2">
        <Button
          variant={featured ? "default" : "outline"}
          size="lg"
          onClick={handleSubscribe}
          disabled={isPending}
        >
          {isPending ? <Loader2 className="h-4 w-4 animate-spin mr-2" /> : <Check className="h-4 w-4 mr-2" />}
          Subscribe
        </Button>
        {error ? (
          <p className="text-sm text-destructive" role="alert">
            {error === "missing_price_id" ? "Pricing not configured — contact support." :
             error === "unknown_tier" ? "Unexpected tier value." :
             error === "checkout_failed" ? "Could not start checkout. Try again." :
             error === "network_error" ? "Network error. Try again." :
             "Something went wrong."}
          </p>
        ) : null}
      </CardFooter>
    </Card>
  );
}
```

Key implementation notes:
- The current `PlanSelectionCard` likely uses a similar `fetch` pattern — port its error-state handling shape verbatim (the canonical errors are: `missing_price_id`, `unknown_tier`, `use_consult_link`, network failures). Do NOT show `use_consult_link` to the user — that error means the client tried to subscribe to Branding, which can only happen via a bug; treat as `checkout_failed`.
- `data-tier={tier}` makes the card targetable in DOM for UAT screenshots/automation.
- The button uses `variant="default"` for the featured (Widget) card and `variant="outline"` for Basic — visual distinction without re-themeing.

**File 3: `app/(shell)/app/billing/_components/tier-grid.tsx`**

```typescript
"use client";

import { useState } from "react";
import { Tabs, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { Badge } from "@/components/ui/badge";
import { SubscribeTierCard } from "./subscribe-tier-card";
import { ConsultTierCard } from "./consult-tier-card";
import type { PriceInterval } from "@/lib/stripe/prices";

export interface TierGridProps {
  basicMonthlyLabel: string;
  basicAnnualMonthlyEquivalentLabel: string;
  basicAnnualTotalLabel: string;
  basicSavingsPct: number;
  widgetMonthlyLabel: string;
  widgetAnnualMonthlyEquivalentLabel: string;
  widgetAnnualTotalLabel: string;
  widgetSavingsPct: number;
  brandingBookingUrl: string;
}

export function TierGrid({
  basicMonthlyLabel, basicAnnualMonthlyEquivalentLabel, basicAnnualTotalLabel, basicSavingsPct,
  widgetMonthlyLabel, widgetAnnualMonthlyEquivalentLabel, widgetAnnualTotalLabel, widgetSavingsPct,
  brandingBookingUrl,
}: TierGridProps) {
  const [interval, setInterval] = useState<PriceInterval>("annual");
  const maxSavings = Math.max(basicSavingsPct, widgetSavingsPct);

  return (
    <div className="space-y-6">
      {/* Global monthly/annual toggle */}
      <div className="flex justify-center">
        <Tabs value={interval} onValueChange={(v) => setInterval(v as PriceInterval)}>
          <TabsList>
            <TabsTrigger value="monthly">Monthly</TabsTrigger>
            <TabsTrigger value="annual" className="gap-2">
              Annual
              {maxSavings > 0 ? <Badge variant="secondary">Save up to {maxSavings}%</Badge> : null}
            </TabsTrigger>
          </TabsList>
        </Tabs>
      </div>

      {/* 3-card grid */}
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6">
        <SubscribeTierCard
          tier="basic"
          interval={interval}
          monthlyLabel={basicMonthlyLabel}
          annualMonthlyEquivalentLabel={basicAnnualMonthlyEquivalentLabel}
          annualTotalLabel={basicAnnualTotalLabel}
          savingsPct={basicSavingsPct}
        />
        <SubscribeTierCard
          tier="widget"
          interval={interval}
          monthlyLabel={widgetMonthlyLabel}
          annualMonthlyEquivalentLabel={widgetAnnualMonthlyEquivalentLabel}
          annualTotalLabel={widgetAnnualTotalLabel}
          savingsPct={widgetSavingsPct}
          featured
        />
        <ConsultTierCard bookingUrl={brandingBookingUrl} />
      </div>
    </div>
  );
}
```

Layout invariants:
- `grid-cols-1 md:grid-cols-3` satisfies the 1024px-no-scroll verification gate (1024px > md breakpoint 768px).
- `gap-6` is consistent with the rest of the app's spacing.
- The default toggle value is `"annual"` — research recommends annual as default since it shows the savings badge (better SaaS pattern).
