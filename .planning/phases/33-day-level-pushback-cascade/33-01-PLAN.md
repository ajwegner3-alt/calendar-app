```typescript
import { TZDate } from "@date-fns/tz";

export interface PushbackBooking {
  id: string;
  start_at: string;          // UTC ISO
  end_at: string;            // UTC ISO
  status: "confirmed";       // narrowed
  booker_first_name: string;
  booker_email: string;
  duration_minutes: number;       // from event_types join
  buffer_after_minutes: number;   // from event_types join
  event_type_id: string;
  event_type_name: string;        // for owner-facing display in 33-02 if needed
  reschedule_token_hash: string;  // 33-03 needs this for the CAS guard pre-fetch
}

export async function getBookingsForPushback(
  supabase: SupabaseClient,
  params: { accountId: string; dateIsoYmd: string; accountTimezone: string },
): Promise<PushbackBooking[]> {
  // Compute UTC bounds for the local-day window in accountTimezone.
  // Pattern: TZDate-based local-midnight -> UTC ISO; matches getAffectedBookings()
  // in app/(shell)/app/availability/_lib/queries.ts (Phase 32-03).
  const localMidnight = new TZDate(`${params.dateIsoYmd}T00:00:00`, params.accountTimezone);
  const dayStartUtc = localMidnight.toISOString();
  const dayEndUtc = new TZDate(localMidnight.getTime() + 24 * 60 * 60 * 1000, params.accountTimezone).toISOString();

  const { data, error } = await supabase
    .from("bookings")
    .select(`
      id,
      start_at,
      end_at,
      status,
      booker_first_name,
      booker_email,
      reschedule_token_hash,
      event_type_id,
      event_types!inner ( id, name, duration_minutes, buffer_after_minutes )
    `)
    .eq("account_id", params.accountId)
    .eq("status", "confirmed")
    .gte("start_at", dayStartUtc)
    .lt("start_at", dayEndUtc)
    .order("start_at", { ascending: true });

  if (error) throw error;
  return (data ?? []).map((row) => ({
    id: row.id,
    start_at: row.start_at,
    end_at: row.end_at,
    status: "confirmed" as const,
    booker_first_name: row.booker_first_name,
    booker_email: row.booker_email,
    reschedule_token_hash: row.reschedule_token_hash,
    event_type_id: row.event_type_id,
    duration_minutes: row.event_types.duration_minutes,
    buffer_after_minutes: row.event_types.buffer_after_minutes,
    event_type_name: row.event_types.name,
  }));
}
```

```typescript
"use server";

import { createClient } from "@/lib/supabase/server";
import { resolveOwnerContext } from "../../availability/_lib/actions-batch-cancel"; // reuse Phase 32 pattern OR inline auth here
import { getBookingsForPushback, type PushbackBooking } from "./queries";

export interface GetBookingsForPushbackResult {
  ok: boolean;
  bookings?: PushbackBooking[];
  error?: string;
}

export async function getBookingsForPushbackAction(input: {
  accountId: string;
  date: string;          // YYYY-MM-DD
  accountTimezone: string;
}): Promise<GetBookingsForPushbackResult> {
  const supabase = await createClient();
  // Auth check: confirm caller owns accountId
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { ok: false, error: "Unauthorized" };

  const { data: account } = await supabase
    .from("accounts").select("owner_user_id").eq("id", input.accountId).single();
  if (account?.owner_user_id !== user.id) return { ok: false, error: "Unauthorized" };

  try {
    const bookings = await getBookingsForPushback(supabase, {
      accountId: input.accountId,
      dateIsoYmd: input.date,
      accountTimezone: input.accountTimezone,
    });
    return { ok: true, bookings };
  } catch (e) {
    return { ok: false, error: e instanceof Error ? e.message : "Failed to load bookings" };
  }
}
```

```typescript
import { PushbackDialog } from "./pushback-dialog";

export interface PushbackDialogProviderProps {
  accountId: string;
  accountTimezone: string;
  todayIsoYmd: string;
  variant?: "header" | "day-section";
  daySectionDate?: string; // when variant === "day-section"
  // For day-section variant: the parent (BookingsDayGroupedView) renders <PushbackDayButton> instances,
  // which need a way to OPEN the same dialog. This provider exports a Context (below) for that.
}

// Context so day-section buttons can open the header dialog
import { createContext, useContext, type ReactNode } from "react";

interface PushbackDialogContextValue {
  open: (initialDate: string) => void;
}
const PushbackDialogCtx = createContext<PushbackDialogContextValue | null>(null);

export function usePushbackDialog(): PushbackDialogContextValue {
  const ctx = useContext(PushbackDialogCtx);
  if (!ctx) throw new Error("usePushbackDialog must be used within PushbackDialogProvider");
  return ctx;
}

export function PushbackDialogProvider({
  accountId, accountTimezone, todayIsoYmd, children,
}: { accountId: string; accountTimezone: string; todayIsoYmd: string; children: ReactNode }) {
  const [open, setOpen] = useState(false);
  const [initialDate, setInitialDate] = useState(todayIsoYmd);

  const ctxValue: PushbackDialogContextValue = {
    open: (date) => { setInitialDate(date); setOpen(true); },
  };

  return (
    <PushbackDialogCtx.Provider value={ctxValue}>
      {children}
      <PushbackDialog
        open={open}
        onOpenChange={setOpen}
        accountId={accountId}
        accountTimezone={accountTimezone}
        initialDate={initialDate}
      />
    </PushbackDialogCtx.Provider>
  );
}

// Header button — primary entry point, opens dialog with today pre-selected
export function PushbackHeaderButton({ todayIsoYmd }: { todayIsoYmd: string }) {
  const { open } = usePushbackDialog();
  return (
    <Button onClick={() => open(todayIsoYmd)} variant="default">
      Pushback
    </Button>
  );
}

// Day-section button — shortcut, opens dialog pre-filled with that date
export function PushbackDaySectionButton({ date }: { date: string }) {
  const { open } = usePushbackDialog();
  return (
    <Button size="sm" variant="outline" onClick={() => open(date)}>
      Pushback this day
    </Button>
  );
}
```

STEP 2 — Create `bookings-day-grouped-view.tsx` — the new view used when filter is "upcoming":

```typescript
import { TZDate } from "@date-fns/tz";
import type { BookingRow } from "../_lib/queries";
import { PushbackDaySectionButton } from "./pushback-dialog-provider";
import { BookingRowDisplay } from "./bookings-table"; // assuming exported; otherwise copy the row render here

export interface BookingsDayGroupedViewProps {
  bookings: BookingRow[];   // already filtered to upcoming + sorted ascending by start_at
  accountTimezone: string;
}

export function BookingsDayGroupedView({ bookings, accountTimezone }: BookingsDayGroupedViewProps) {
  // Group by local-day in accountTimezone
  const groups = new Map<string, BookingRow[]>();
  for (const b of bookings) {
    const local = new TZDate(new Date(b.start_at), accountTimezone);
    const key = `${local.getFullYear()}-${String(local.getMonth() + 1).padStart(2, "0")}-${String(local.getDate()).padStart(2, "0")}`;
    if (!groups.has(key)) groups.set(key, []);
    groups.get(key)!.push(b);
  }
  const orderedKeys = [...groups.keys()].sort();

  if (orderedKeys.length === 0) {
    return <div className="py-8 text-center text-sm text-muted-foreground">No upcoming bookings.</div>;
  }

  return (
    <div className="space-y-6">
      {orderedKeys.map((dayKey) => {
        const dayBookings = groups.get(dayKey)!;
        const friendlyLabel = formatDayLabel(dayKey, accountTimezone); // e.g., "Mon, May 6"
        return (
          <section key={dayKey} aria-label={friendlyLabel}>
            <header className="flex items-center justify-between mb-2 pb-1 border-b">
              <h2 className="text-sm font-semibold">{friendlyLabel}</h2>
              <PushbackDaySectionButton date={dayKey} />
            </header>
            <ul className="divide-y">
              {dayBookings.map((b) => (
                <li key={b.id} className="py-2">
                  {/* Reuse the existing row render — extract from bookings-table.tsx as <BookingRowDisplay> if it isn't already exported */}
                  <BookingRowDisplay booking={b} accountTimezone={accountTimezone} />
                </li>
              ))}
            </ul>
          </section>
        );
      })}
    </div>
  );
}

function formatDayLabel(ymd: string, tz: string): string {
  const [y, m, d] = ymd.split("-").map(Number);
  const local = new TZDate(new Date(Date.UTC(y, m - 1, d)), tz);
  return new Intl.DateTimeFormat("en-US", {
    weekday: "short", month: "short", day: "numeric", timeZone: tz,
  }).format(local);
}
```
