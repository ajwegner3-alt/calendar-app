```typescript
"use server";

import { revalidatePath } from "next/cache";
import { generateBookingTokens, hashToken } from "@/lib/bookings/tokens";
import { sendRescheduleEmails } from "@/lib/email/send-reschedule-emails";
// ... existing imports ...

export interface RetryPushbackEmailInput {
  accountId: string;
  bookingId: string;
  oldStartAt: string;        // From the commit response — needed for the "Was:" line in the email
  reason?: string;           // Same reason text used in the original batch
}

export type RetryPushbackEmailResult =
  | { ok: true }
  | { ok: false; quotaError: true; remaining: number }
  | { ok: false; error: string };

export async function retryPushbackEmailAction(input: RetryPushbackEmailInput): Promise<RetryPushbackEmailResult> {
  const supabase = await createClient();

  // 1. AUTH
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { ok: false, error: "Unauthorized" };
  const { data: account } = await supabase
    .from("accounts").select("owner_user_id, timezone").eq("id", input.accountId).single();
  if (account?.owner_user_id !== user.id) return { ok: false, error: "Unauthorized" };

  // 2. Pre-flight quota: 1 booker email
  const remaining = await getRemainingDailyQuota();
  if (remaining < 1) return { ok: false, quotaError: true, remaining };

  // 3. Re-fetch the booking (current state — start_at is now the NEW time from the commit)
  const { data: booking, error: fetchErr } = await supabase
    .from("bookings")
    .select(`
      id, account_id, event_type_id,
      booker_first_name, booker_email,
      start_at, end_at, status,
      event_types!inner ( id, name, slug, duration_minutes )
    `)
    .eq("id", input.bookingId)
    .eq("account_id", input.accountId)
    .single();
  if (fetchErr || !booking) return { ok: false, error: "Booking not found" };
  if (booking.status !== "confirmed") return { ok: false, error: "Booking is not confirmed" };

  // 4. Mint fresh tokens (matches the reminder retry idiom — invalidates any older raw tokens)
  const { rawCancelToken, rawRescheduleToken } = generateBookingTokens();
  const cancelHash = hashToken(rawCancelToken);
  const rescheduleHash = hashToken(rawRescheduleToken);
  const { error: updateErr } = await supabase
    .from("bookings")
    .update({
      cancellation_token_hash: cancelHash,
      reschedule_token_hash: rescheduleHash,
    })
    .eq("id", booking.id);
  if (updateErr) return { ok: false, error: "Failed to refresh tokens: " + updateErr.message };

  // 5. Send the booker leg only (sendOwner=false; LD-07 booker-neutrality preserved)
  try {
    await sendRescheduleEmails({
      // Match the exact arg shape that sendRescheduleEmails expects — verify against the file from 33-03.
      bookingId: booking.id,
      bookerEmail: booking.booker_email,
      bookerFirstName: booking.booker_first_name,
      eventTypeName: booking.event_types.name,
      oldStartAt: input.oldStartAt,
      newStartAt: booking.start_at,
      newEndAt: booking.end_at,
      durationMinutes: booking.event_types.duration_minutes,
      rawCancelToken,
      rawRescheduleToken,
      reason: input.reason,
      sendOwner: false,                    // owner already saw the in-dialog summary; don't re-notify
      // ...any other required args (account branding, slug, etc.)...
    });
  } catch (e) {
    const code = (e as { code?: string }).code;
    if (code === "EMAIL_QUOTA_EXCEEDED") {
      const remaining = await getRemainingDailyQuota();
      return { ok: false, quotaError: true, remaining };
    }
    return { ok: false, error: e instanceof Error ? e.message : "Email send failed" };
  }

  revalidatePath("/app/bookings");
  return { ok: true };
}
```

The exact `sendRescheduleEmails` arg shape will depend on what 33-03 finalized — read the file's signature and adjust the call accordingly. The key invariants are: `sendOwner: false`, `oldStartAt` from the input (NOT from the booking row, since the booking is now at the NEW time), and the freshly-minted raw tokens.

STEP 2 — Add a `RetryEmailButton` component scoped to the dialog file:

```tsx
function RetryEmailButton({
  row, accountId, reason, onSuccess,
}: {
  row: CommitPushbackResultRow;
  accountId: string;
  reason: string;
  onSuccess: (bookingId: string) => void;
}) {
  const [pending, startTransition] = useTransition();
  const handleRetry = () => {
    startTransition(async () => {
      const result = await retryPushbackEmailAction({
        accountId,
        bookingId: row.booking_id,
        oldStartAt: row.old_start_at,
        reason: reason || undefined,
      });
      if (result.ok) {
        onSuccess(row.booking_id);
        toast.success(`Email sent to ${row.booker_first_name}`);
        return;
      }
      if ("quotaError" in result && result.quotaError) {
        toast.error(`Quota exhausted (${result.remaining} remaining). Try again tomorrow.`);
        return;
      }
      toast.error("error" in result ? result.error : "Retry failed");
    });
  };
  return (
    <Button size="sm" variant="outline" disabled={pending} onClick={handleRetry}>
      {pending ? "Retrying…" : "Retry email"}
    </Button>
  );
}
```
