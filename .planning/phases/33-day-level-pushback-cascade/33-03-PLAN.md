"use server";

import { revalidatePath } from "next/cache";
import { rescheduleBooking } from "@/lib/bookings/reschedule";
import { getRemainingDailyQuota } from "@/lib/email-sender/quota-guard";
import { getBookingsForPushback } from "./queries";
// ... existing imports ...

export interface CommitPushbackInput {
  accountId: string;
  date: string;                 // YYYY-MM-DD in accountTimezone
  accountTimezone: string;
  reason?: string;
  // From the preview (preview-approved set):
  movedBookings: Array<{
    booking_id: string;
    new_start_at: string;       // UTC ISO from CascadeRow.new_start_at
    new_end_at: string;
  }>;
  // The full list of confirmed-booking IDs the preview saw, including ABSORBED ones.
  // Used for the abort-on-diverge re-query comparison.
  previewBookingIds: string[];
}

export interface CommitPushbackResultRow {
  booking_id: string;
  booker_first_name: string;
  old_start_at: string;
  new_start_at: string | null;            // null if not moved
  status: "sent" | "email_failed" | "slot_taken" | "not_active" | "skipped";
  error_message?: string;                 // when status !== 'sent' (or 'skipped' for ABSORBED)
}

export type CommitPushbackResult =
  | { ok: true; rows: CommitPushbackResultRow[] }
  | { ok: false; quotaError: true; needed: number; remaining: number }
  | { ok: false; diverged: true; message: string }
  | { ok: false; formError: string };

export async function commitPushbackAction(input: CommitPushbackInput): Promise<CommitPushbackResult> {
  const supabase = await createClient();

  // 1. AUTH: caller owns accountId
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { ok: false, formError: "Unauthorized" };
  const { data: account } = await supabase
    .from("accounts").select("owner_user_id").eq("id", input.accountId).single();
  if (account?.owner_user_id !== user.id) return { ok: false, formError: "Unauthorized" };

  // 2. HARD QUOTA PRE-FLIGHT (re-checked at commit, not just preview)
  const remaining = await getRemainingDailyQuota();
  const needed = input.movedBookings.length; // skipOwnerEmail=true → 1 booker email per moved booking
  if (needed > remaining) {
    return { ok: false, quotaError: true, needed, remaining };
  }

  // 3. RACE-SAFE RE-QUERY: re-fetch the day's confirmed bookings RIGHT NOW
  const currentBookings = await getBookingsForPushback(supabase, {
    accountId: input.accountId, dateIsoYmd: input.date, accountTimezone: input.accountTimezone,
  });
  const currentIds = new Set(currentBookings.map((b) => b.id));
  const previewIds = new Set(input.previewBookingIds);

  // 4. ABORT ON DIVERGE: any addition or removal aborts the commit
  const sameSet = currentIds.size === previewIds.size
    && [...previewIds].every((id) => currentIds.has(id));
  if (!sameSet) {
    return {
      ok: false, diverged: true,
      message: "Bookings on this date changed since preview. Review again.",
    };
  }

  // 5. PRE-FETCH reschedule_token_hash for every moving booking (needed for rescheduleBooking CAS guard)
  const movingIds = input.movedBookings.map((m) => m.booking_id);
  const movingBookingsMap = new Map(currentBookings.filter((b) => movingIds.includes(b.id)).map((b) => [b.id, b]));

  // 6. BATCH RESCHEDULE: Promise.allSettled per moving booking
  const results: CommitPushbackResultRow[] = [];

  // ABSORBED bookings: emit "skipped" rows so the summary lists every booking
  for (const id of input.previewBookingIds) {
    if (movingIds.includes(id)) continue; // moving rows added below
    const b = currentBookings.find((x) => x.id === id);
    if (!b) continue; // shouldn't happen given the diverge check, but safety
    results.push({
      booking_id: id,
      booker_first_name: b.booker_first_name,
      old_start_at: b.start_at,
      new_start_at: null,
      status: "skipped",
    });
  }

  // Moving rows
  const movingResults = await Promise.allSettled(
    input.movedBookings.map(async (m) => {
      const b = movingBookingsMap.get(m.booking_id);
      if (!b) {
        return {
          booking_id: m.booking_id, booker_first_name: "?",
          old_start_at: "?", new_start_at: m.new_start_at,
          status: "not_active" as const, error_message: "Booking not found at commit",
        };
      }
      try {
        await rescheduleBooking({
          bookingId: b.id,
          newStartAt: m.new_start_at,
          newEndAt: m.new_end_at,
          oldRescheduleHash: b.reschedule_token_hash,
          reason: input.reason,
          skipOwnerEmail: true,        // Phase 33 lock
          actor: "owner",              // Phase 33 lock
          // ... any other required RescheduleBookingArgs (cancellationToken regen handled inside) ...
        });
        return {
          booking_id: b.id, booker_first_name: b.booker_first_name,
          old_start_at: b.start_at, new_start_at: m.new_start_at,
          status: "sent" as const,
        };
      } catch (e) {
        // rescheduleBooking() throws with a `code` field per RESEARCH.md (slot_taken/not_active/bad_slot/db_error)
        const code = (e as { code?: string }).code;
        const message = e instanceof Error ? e.message : "Unknown error";
        const status: CommitPushbackResultRow["status"] =
          code === "slot_taken" ? "slot_taken"
          : code === "not_active" ? "not_active"
          : code === "email_failed" ? "email_failed" // if rescheduleBooking surfaces email failures with this code; else use a wrapper
          : "email_failed";   // safe default for an unmapped failure that still UPDATEd the row
        return {
          booking_id: b.id, booker_first_name: b.booker_first_name,
          old_start_at: b.start_at,
          new_start_at: status === "slot_taken" || status === "not_active" ? null : m.new_start_at,
          status, error_message: message,
        };
      }
    })
  );

  for (const settled of movingResults) {
    if (settled.status === "fulfilled") {
      results.push(settled.value);
    } else {
      // unexpected throw outside the try/catch — should not happen, but be defensive
      results.push({
        booking_id: "?", booker_first_name: "?",
        old_start_at: "?", new_start_at: null,
        status: "email_failed", error_message: String(settled.reason),
      });
    }
  }

  // 7. REVALIDATE the bookings page so the day-grouped view refreshes
  revalidatePath("/app/bookings");

  return { ok: true, rows: results };
}
```

STEP 3 — Important nuance — distinguishing email_failed from slot_taken (RESEARCH.md Risk 7):
- `slot_taken` (DB constraint violation): the UPDATE failed; booking is NOT changed. Booker received NO email. Owner cannot retry email — the new time is taken.
- `not_active` (CAS hash mismatch): the UPDATE failed; booking might be cancelled or already-rescheduled by another path. Owner cannot retry email.
- `email_failed`: the UPDATE succeeded; booking IS changed; but the email did not send. Owner CAN retry email (33-04 wires the retry button).

`rescheduleBooking()` currently bundles UPDATE + email into a single function. To cleanly distinguish, look at how rescheduleBooking surfaces email failures (RESEARCH.md notes it throws on email errors but the booking row is already updated). A. Add a try/catch around the email-send call inside `rescheduleBooking()` so DB-success-email-failure becomes a distinct return shape (e.g., `{ ok: true, emailFailed: true }`).

OR

B. Wrap `rescheduleBooking()` in a thin helper inside `actions-pushback.ts` that splits the UPDATE and email-send into two phases.

    const previewBookingIds = previewRows.map((r) => r.booking.id);

    const result = await commitPushbackAction({
      accountId, date, accountTimezone, reason: reason || undefined,
      movedBookings, previewBookingIds,
    });

    if ("quotaError" in result && result.quotaError) {
      // Drift between preview and commit — quota was consumed by another path
      setRemainingQuota(result.remaining);
      setQuotaError(true);
      setState("preview-ready");
      toast.error(`Quota changed: ${result.needed} needed, ${result.remaining} remaining.`);
      return;
    }

    if ("diverged" in result && result.diverged) {
      setDivergedMessage(result.message);
      setState("editing");
      // Re-fetch bookings to refresh the list (the date hasn't changed)
      // Easiest: trigger a fetch by toggling an effect or directly calling getBookingsForPushbackAction
      const refresh = await getBookingsForPushbackAction({ accountId, date, accountTimezone });
      if (refresh.ok) {
        setBookings(refresh.bookings);
        setAnchorId(refresh.bookings[0]?.id ?? null);
      }
      toast.error(result.message);
      return;
    }

    if (!result.ok) {
      toast.error("formError" in result ? result.formError : "Failed to commit pushback");
      setState("preview-ready");
      return;
    }

    // Success — store rows and transition to summary
    setCommitRows(result.rows);
    setState("summary");

    const sentCount = result.rows.filter((r) => r.status === "sent").length;
    const failureCount = result.rows.filter((r) => r.status === "email_failed" || r.status === "slot_taken" || r.status === "not_active").length;
    if (failureCount === 0) {
      toast.success(`${sentCount} pushback email${sentCount === 1 ? "" : "s"} sent.`);
    } else {
      toast.warning(`${sentCount} sent · ${failureCount} need attention. See summary.`);
    }
  });
}
```
