```typescript
/**
 * Returns the local-time end-of-day minute for a date.
 * - 1440 (24*60) = "no constraint" (used when the day has no availability rules and no override)
 *                   matches CONTEXT.md OQ-5 decision: no-rules days never flag PAST_EOD
 * - Otherwise: the maximum end_minute across all available windows for that date
 *
 * Inputs match windowsForDate(); this is a thin wrapper that collapses the
 * windows array to a single end-of-day scalar suitable for cascade PAST_EOD checks.
 */
export function getEndOfDayMinute(
  dateIsoYmd: string,                    // YYYY-MM-DD (local date)
  dayOfWeek: number,                      // 0..6
  rules: AvailabilityRuleRow[],           // weekly availability for the account
  overrides: DateOverrideRow[],           // date overrides for the account (Phase 32 shape)
): number {
  const windows = windowsForDate(dateIsoYmd, dayOfWeek, rules, overrides);
  if (windows === null) {
    // Closed day per rules+overrides — but per CONTEXT.md OQ-5, treat as no-constraint (1440)
    // so the dialog allows ad-hoc bookings on otherwise-closed days without flagging PAST_EOD.
    return 24 * 60;
  }
  if (windows.length === 0) return 24 * 60;
  return Math.max(...windows.map((w) => w.end_minute));
}
```

```typescript
import { TZDate } from "@date-fns/tz";

export type CascadeStatus = "MOVE" | "ABSORBED" | "PAST_EOD";

export interface PushbackBookingInput {
  id: string;
  start_at: string;             // UTC ISO
  end_at: string;               // UTC ISO
  booker_first_name: string;
  duration_minutes: number;     // event_type.duration_minutes
  buffer_after_minutes: number; // event_type.buffer_after_minutes
  event_type_id: string;
}

export interface CascadeRow {
  booking: PushbackBookingInput;
  status: CascadeStatus;
  old_start_at: string;
  new_start_at: string | null;  // null if ABSORBED
  new_end_at: string | null;
}

export interface ComputeCascadeArgs {
  bookings: PushbackBookingInput[]; // ALL confirmed bookings for the date, sorted by start_at ASC
  anchorId: string;                  // first booking that must move
  delayMs: number;                   // delay in milliseconds
  endOfDayMinutes: number;           // 0..1440; 1440 means "no constraint"
  accountTimezone: string;           // IANA tz, e.g. "America/Chicago"
}

/**
 * Pure cascade preview.
 *
 * Algorithm (CONTEXT.md lock):
 * - Bookings strictly before the anchor are untouched (status: ABSORBED with new_start_at=null).
 * - The anchor moves by exactly delayMs, then snap UP to the anchor's own duration_minutes grid.
 * - For each subsequent booking i:
 *     candidateStart = priorNewEnd + bookings[i-1].buffer_after_minutes * 60_000
 *     if candidateStart <= bookings[i].original_start: ABSORBED (priorNewEnd resets to bookings[i].original_end)
 *     else: snap candidateStart UP to bookings[i].duration_minutes grid; classify MOVE or PAST_EOD
 *
 * PAST_EOD: local-time minutes-since-midnight of the new start >= endOfDayMinutes
 *           (1440 sentinel disables this check entirely)
 */
export function computeCascadePreview(args: ComputeCascadeArgs): CascadeRow[] {
  const { bookings, anchorId, delayMs, endOfDayMinutes, accountTimezone } = args;
  const anchorIdx = bookings.findIndex((b) => b.id === anchorId);
  if (anchorIdx < 0) throw new Error("Anchor booking not found in bookings list");

  const result: CascadeRow[] = [];

  // Before-anchor: untouched
  for (let i = 0; i < anchorIdx; i++) {
    const b = bookings[i];
    result.push({
      booking: b, status: "ABSORBED",
      old_start_at: b.start_at, new_start_at: null, new_end_at: null,
    });
  }

  // At/after-anchor: cascade
  let priorNewEndMs = 0;
  for (let i = anchorIdx; i < bookings.length; i++) {
    const b = bookings[i];
    const origStartMs = new Date(b.start_at).getTime();
    const origEndMs = new Date(b.end_at).getTime();
    const durationMs = origEndMs - origStartMs;

    let candidateStartMs: number;
    if (i === anchorIdx) {
      candidateStartMs = origStartMs + delayMs;
    } else {
      const priorBufferMs = bookings[i - 1].buffer_after_minutes * 60_000;
      candidateStartMs = priorNewEndMs + priorBufferMs;
    }

    if (i > anchorIdx && candidateStartMs <= origStartMs) {
      // ABSORBED — gap absorbs the push
      priorNewEndMs = origEndMs; // cascade frontier resets to this booking's original end
      result.push({
        booking: b, status: "ABSORBED",
        old_start_at: b.start_at, new_start_at: null, new_end_at: null,
      });
      continue;
    }

    // MOVE or PAST_EOD
    const snappedStartMs = snapToNextSlotMs(candidateStartMs, b.duration_minutes);
    const newEndMs = snappedStartMs + durationMs;
    priorNewEndMs = newEndMs;
    const status: CascadeStatus = isPastEod(snappedStartMs, accountTimezone, endOfDayMinutes)
      ? "PAST_EOD" : "MOVE";

    result.push({
      booking: b, status,
      old_start_at: b.start_at,
      new_start_at: new Date(snappedStartMs).toISOString(),
      new_end_at: new Date(newEndMs).toISOString(),
    });
  }

  return result;
}

export function snapToNextSlotMs(rawMs: number, slotStepMinutes: number): number {
  const stepMs = slotStepMinutes * 60_000;
  return Math.ceil(rawMs / stepMs) * stepMs;
}

export function isPastEod(startMs: number, accountTimezone: string, endOfDayMinutes: number): boolean {
  if (endOfDayMinutes >= 24 * 60) return false; // 1440 sentinel: no constraint
  const local = new TZDate(new Date(startMs), accountTimezone);
  const localMinutes = local.getHours() * 60 + local.getMinutes();
  return localMinutes >= endOfDayMinutes;
}

/** Helper for callers that want quick counts. */
export function countMoved(rows: CascadeRow[]): number {
  return rows.filter((r) => r.status === "MOVE" || r.status === "PAST_EOD").length;
}
```

```typescript
import { createClient } from "@/lib/supabase/server";
import { getBookingsForPushback } from "./queries";
import { getEndOfDayMinute } from "@/lib/slots";
import { computeCascadePreview, countMoved, type CascadeRow } from "@/lib/bookings/pushback";
import { getRemainingDailyQuota } from "@/lib/email-sender/quota-guard";
import { TZDate } from "@date-fns/tz";

export interface PreviewPushbackInput {
  accountId: string;
  date: string;             // YYYY-MM-DD in accountTimezone
  accountTimezone: string;
  anchorId: string;
  delayMinutes: number;     // already converted from min/hr in the dialog
  reason?: string;
}

export type PreviewPushbackResult =
  | {
      ok: true;
      rows: CascadeRow[];
      movedCount: number;          // MOVE + PAST_EOD count (= emails to send)
      remainingQuota: number;
      quotaError: boolean;          // true if movedCount > remainingQuota
    }
  | { ok: false; error: string };

export async function previewPushbackAction(input: PreviewPushbackInput): Promise<PreviewPushbackResult> {
  const supabase = await createClient();
  const { data: { user } } = await supabase.auth.getUser();
  if (!user) return { ok: false, error: "Unauthorized" };

  const { data: account } = await supabase
    .from("accounts").select("owner_user_id").eq("id", input.accountId).single();
  if (account?.owner_user_id !== user.id) return { ok: false, error: "Unauthorized" };

  // Validate delay
  if (!Number.isInteger(input.delayMinutes) || input.delayMinutes <= 0) {
    return { ok: false, error: "Delay must be a positive integer" };
  }

  // Fetch bookings for the date
  const bookings = await getBookingsForPushback(supabase, {
    accountId: input.accountId, dateIsoYmd: input.date, accountTimezone: input.accountTimezone,
  });
  if (bookings.length === 0) return { ok: false, error: "No bookings on this date" };

  // Fetch availability rules + date overrides for endOfDayMinute computation
  const [{ data: rules }, { data: overrides }] = await Promise.all([
    supabase.from("availability_rules").select("*").eq("account_id", input.accountId),
    supabase.from("date_overrides").select("*").eq("account_id", input.accountId).eq("override_date", input.date),
  ]);

  // Day-of-week for the chosen date in accountTimezone
  const localMidnight = new TZDate(`${input.date}T00:00:00`, input.accountTimezone);
  const dow = localMidnight.getDay();

  const endOfDayMinutes = getEndOfDayMinute(input.date, dow, rules ?? [], overrides ?? []);

  const rows = computeCascadePreview({
    bookings,
    anchorId: input.anchorId,
    delayMs: input.delayMinutes * 60_000,
    endOfDayMinutes,
    accountTimezone: input.accountTimezone,
  });

  const movedCount = countMoved(rows);
  const remainingQuota = await getRemainingDailyQuota();

  return {
    ok: true,
    rows,
    movedCount,
    remainingQuota,
    quotaError: movedCount > remainingQuota,
  };
}
```
